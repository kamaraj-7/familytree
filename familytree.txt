        // Calculate age
        function calculateAge(dob) {
            const birthDate = new Date(dob);
            const today = new Date();
            let age = today.getFullYear() - birthDate.getFullYear();
            const monthDiff = today.getMonth() - birthDate.getMonth();
            if (monthDiff < 0 || (monthDiff === 0 && today.getDate() < birthDate.getDate())) {
                age--;
            }
            return age;
        }

        // Check if birthday is today
        function isBirthdayToday(dob) {
            const birthDate = new Date(dob);
            const today = new Date();
            return birthDate.getDate() === today.getDate() && 
                   birthDate.getMonth() === today.getMonth();
        }

        // Calculate days until birthday
        function daysUntilBirthday(dob) {
            const birthDate = new Date(dob);
            const today = new Date();
            const currentYear = today.getFullYear();
            
            let nextBirthday = new Date(currentYear, birthDate.getMonth(), birthDate.getDate());
            
            if (nextBirthday < today) {
                nextBirthday = new Date(currentYear + 1, birthDate.getMonth(), birthDate.getDate());
            }
            
            const diffTime = nextBirthday - today;
            const diffDays = Math.ceil(diffTime / (1000 * 60 * 60 * 24));
            return diffDays;
        }

            `;
            
            modal.classList.add('active');
        }

        // Get ordinal suffix
        function getOrdinalSuffix(num) {
            const j = num % 10;
            const k = num % 100;
            if (j === 1 && k !== 11) return "st";
            if (j === 2 && k !== 12) return "nd";
            if (j === 3 && k !== 13) return "rd";
            return "th";
        }

        // Render family tree
        function renderFamilyTree(members = familyMembers) {
            const treeContainer = document.getElementById('familyTree');
            const emptyState = document.getElementById('emptyState');
            
            treeContainer.innerHTML = '';
            
            if (members.length === 0) {
                emptyState.style.display = 'block';
                treeContainer.style.display = 'none';
            } else {
                emptyState.style.display = 'none';
                treeContainer.style.display = 'grid';
                members.forEach(member => {
                    treeContainer.appendChild(renderMemberCard(member));
                });
            }
        }

        // Update birthday banner
        function updateBirthdayBanner() {
            const todaysBirthdays = familyMembers.filter(m => isBirthdayToday(m.dob));
            const banner = document.getElementById('birthdayBanner');
            const celebrantsList = document.getElementById('celebrantsList');
            
            if (todaysBirthdays.length > 0) {
                banner.classList.add('active');
                celebrantsList.innerHTML = todaysBirthdays.map(m => `
                    <div class="celebrant-card">
                        <strong>${m.name}</strong><br>
                        <small>${calculateAge(m.dob)} years old</small>
                    </div>
                `).join('');
            } else {
                banner.classList.remove('active');
            }
        }

        // Update upcoming birthdays
        function updateUpcomingBirthdays() {
            const container = document.getElementById('upcomingBirthdays');
            const upcoming = familyMembers
                .map(m => ({...m, daysUntil: daysUntilBirthday(m.dob)}))
                .filter(m => m.daysUntil > 0 && m.daysUntil <= 60)
                .sort((a, b) => a.daysUntil - b.daysUntil)
                .slice(0, 6);
            
            container.innerHTML = upcoming.length > 0 ? upcoming.map(m => `
                <div class="countdown-card">
                    <img src="${m.photo || 'https://via.placeholder.com/80'}" alt="${m.name}" style="width: 80px; height: 80px; border-radius: 50%; margin-bottom: 15px;">
                    <h4>${m.name}</h4>
                    <div class="countdown-timer">${m.daysUntil} days</div>
                    <p>${new Date(m.dob).toLocaleDateString('en-US', {month: 'long', day: 'numeric'})}</p>
                </div>
            `).join('') : '<p style="text-align: center; color: var(--text-light);">No upcoming birthdays in the next 60 days</p>';
        }

        // Load family members
        async function loadFamilyMembers() {
            if (!db) return;
            
            document.getElementById('loading').classList.add('active');
            
            const q = query(collection(db, 'familyMembers'), orderBy('name'));
            onSnapshot(q, (snapshot) => {
                familyMembers = [];
                snapshot.forEach((doc) => {
                    familyMembers.push({ id: doc.id, ...doc.data() });
                });
                
                renderFamilyTree();
                updateBirthdayBanner();
                updateUpcomingBirthdays();
                document.getElementById('loading').classList.remove('active');
            });
        }

        // Add or update member
        async function saveMember(e) {
            e.preventDefault();
            
            if (!db) {
                alert('Firebase not configured. Please add your Firebase credentials.');
                return;
            }
            
            const memberData = {
                name: document.getElementById('name').value,
                dob: document.getElementById('dob').value,
                photo: document.getElementById('photo').value || '',
                gender: document.getElementById('gender').value,
                relationship: document.getElementById('relationship').value,
                phone: document.getElementById('phone').value || '',
                notes: document.getElementById('notes').value || '',
                updatedAt: new Date().toISOString()
            };
            
            try {
                if (editingId) {
                    await updateDoc(doc(db, 'familyMembers', editingId), memberData);
                    alert('Family member updated successfully!');
                    editingId = null;
                    document.getElementById('cancelEdit').style.display = 'none';
                } else {
                    memberData.createdAt = new Date().toISOString();
                    await addDoc(collection(db, 'familyMembers'), memberData);
                    alert('Family member added successfully!');
                }
                
                document.getElementById('memberForm').reset();
            } catch (error) {
                console.error('Error saving member:', error);
                alert('Error saving family member. Please try again.');
            }
        }

        // Edit member
        window.editMember = async function(id) {
            const member = familyMembers.find(m => m.id === id);
            if (!member) return;
            
            editingId = id;
            document.getElementById('name').value = member.name;
            document.getElementById('dob').value = member.dob;
            document.getElementById('photo').value = member.photo || '';
            document.getElementById('gender').value = member.gender || '';
            document.getElementById('relationship').value = member.relationship || '';
            document.getElementById('phone').value = member.phone || '';
            document.getElementById('notes').value = member.notes || '';
            document.getElementById('cancelEdit').style.display = 'inline-block';
            
            window.scrollTo({ top: 0, behavior: 'smooth' });
        };

        // Delete member
        window.deleteMember = async function(id) {
            if (!confirm('Are you sure you want to delete this family member?')) return;
            
            try {
                await deleteDoc(doc(db, 'familyMembers', id));
                alert('Family member deleted successfully!');
            } catch (error) {
                console.error('Error deleting member:', error);
                alert('Error deleting family member. Please try again.');
            }
        };

        // Show birthday modal by ID
        window.showBirthdayModal = function(id) {
            const member = familyMembers.find(m => m.id === id);
            if (member) {
                showBirthdayModal(member);
            }
        };

        // Search and filter
        function filterMembers() {
            const searchTerm = document.getElementById('searchInput').value.toLowerCase();
            const relationshipFilter = document.getElementById('filterRelationship').value;
            
            let filtered = familyMembers;
            
            if (searchTerm) {
                filtered = filtered.filter(m => 
                    m.name.toLowerCase().includes(searchTerm) ||
                    (m.notes && m.notes.toLowerCase().includes(searchTerm))
                );
            }
            
            if (relationshipFilter) {
                filtered = filtered.filter(m => 
                    m.relationship && m.relationship.includes(relationshipFilter)
                );
            }
            
            renderFamilyTree(filtered);
        }

        // Event listeners
        document.getElementById('memberForm').addEventListener('submit', saveMember);
        document.getElementById('searchInput').addEventListener('input', filterMembers);
        document.getElementById('filterRelationship').addEventListener('change', filterMembers);
        
        document.getElementById('cancelEdit').addEventListener('click', () => {
            editingId = null;
            document.getElementById('memberForm').reset();
            document.getElementById('cancelEdit').style.display = 'none';
        });
        
        document.getElementById('closeModal').addEventListener('click', () => {
            document.getElementById('birthdayModal').classList.remove('active');
        });
        
        document.getElementById('birthdayModal').addEventListener('click', (e) => {
            if (e.target === document.getElementById('birthdayModal')) {
                document.getElementById('birthdayModal').classList.remove('active');
            }
        });

        // Check for birthdays every hour
        setInterval(() => {
            updateBirthdayBanner();
            updateUpcomingBirthdays();
